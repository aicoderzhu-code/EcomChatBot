```python
from enum import Enum

class QuotaType(Enum):
    CONVERSATION = "conversation"  # 对话次数
    API_CALL = "api_call"          # API调用次数
    STORAGE = "storage"            # 存储空间
    CONCURRENT = "concurrent"      # 并发会话

class OverLimitStrategy(Enum):
    REJECT = "reject"              # 拒绝服务
    UPGRADE_PROMPT = "upgrade"     # 提示升级
    PAY_AS_YOU_GO = "payg"         # 按量付费

def check_quota(quota_type: QuotaType, amount: int = 1):
    """
    配额检查装饰器
    用法: @check_quota(QuotaType.CONVERSATION)
    """
    def decorator(func):
        async def wrapper(*args, **kwargs):
            tenant_id = get_current_tenant_id()
            result = await quota_service.check_and_consume(
                tenant_id, quota_type, amount
            )
            if not result.allowed:
                return handle_over_limit(result.strategy)
            return await func(*args, **kwargs)
        return wrapper
    return decorator
```

```python
def build_conversation_graph() -> StateGraph:

    # 添加节点
    graph.add_node("intent", intent_recognition_node)
    graph.add_node("retrieve", knowledge_retrieval_node)
```

```python
# B2.1 月度账单生成
# 文件: backend/tasks/billing_tasks.py

@celery_app.task(bind=True, max_retries=3)
def generate_monthly_bills(self, billing_period: str = None):
    """
    生成月度账单
    - 每月1日凌晨2点执行
    - 计算上月用量
    - 生成账单记录
    - 发送账单通知
    """
    if billing_period is None:
        billing_period = get_last_month()

    tenants = get_active_tenants()

    for tenant in tenants:
        try:
            # 1. 获取订阅信息
            subscription = get_subscription(tenant.id)

            # 2. 计算基础费用
            base_fee = subscription.plan.monthly_price

            # 3. 计算超额费用
            usage = get_monthly_usage(tenant.id, billing_period)
            overage_fee = calculate_overage(usage, subscription)

            # 4. 应用折扣
            discount = calculate_discount(tenant)

            # 5. 生成账单
            bill = Bill(
                tenant_id=tenant.id,
                period=billing_period,
                base_fee=base_fee,
                overage_fee=overage_fee,
                discount=discount,
                total_amount=base_fee + overage_fee - discount,
                status="pending"
            )
            db.add(bill)

            # 6. 发送通知
            send_bill_notification.delay(tenant.id, bill.id)

        except Exception as e:
            logger.error(f"Failed to generate bill for tenant {tenant.id}: {e}")
            self.retry(exc=e, countdown=60)
```

```python
# C2.2 完整RAG流程
# 文件: backend/services/rag_service.py

class RAGService:
    async def process(
        self,
        tenant_id: str,
        query: str,
        conversation_context: list = None
    ) -> RAGResult:
        """
        完整的7步RAG流程:
        1. 权限验证 - 验证租户权限和配额
        2. 查询预处理 - 扩展、改写查询
        3. 向量化 - 将查询转为向量
        4. 检索 - 从Milvus检索相关文档
        5. 重排序 - Rerank提升相关性
        6. Prompt构建 - 组装上下文和查询
        7. LLM生成 - 调用大模型生成回答
        8. 后处理 - 安全检查、格式化
        """

        # Step 1: 权限验证
        await self._verify_permission(tenant_id)

        # Step 2: 查询预处理
        processed_query = await self._preprocess_query(query)

        # Step 3: 向量化
        query_vector = await self.embedding_service.embed(processed_query)

        # Step 4: 检索 (租户隔离)
        documents = await self.milvus_service.search(
            collection=f"tenant_{tenant_id}",
            vector=query_vector,
            top_k=10
        )

        # Step 5: 重排序
        reranked_docs = await self.rerank_service.rerank(
            query=processed_query,
            documents=documents,
            top_k=5
        )

        # Step 6: Prompt构建
        prompt = self._build_prompt(
            query=query,
            context=conversation_context,
            documents=reranked_docs
        )

        # Step 7: LLM生成
        response = await self.llm_service.generate(prompt)

        # Step 8: 后处理
        safe_response = await self._postprocess(response)

        # 记录用量
        await self._record_usage(tenant_id, prompt, response)

        return RAGResult(
            answer=safe_response,
            sources=reranked_docs,
            tokens_used=self._count_tokens(prompt, response)
        )
```

```python
from abc import ABC, abstractmethod

class LLMAdapter(ABC):
    @abstractmethod

        pass

    @abstractmethod
    async def stream_generate(self, messages: list, **kwargs):
        pass

    @abstractmethod
    def count_tokens(self, text: str) -> int:
        pass

# 文件: backend/services/llm/adapters/openai_adapter.py
class OpenAIAdapter(LLMAdapter):
    """OpenAI GPT系列"""
    pass

    pass

# C3.2 模型路由器
# 文件: backend/services/llm/router.py

class ModelRouter:
    """
    根据条件路由到不同模型:
    - 租户配置: 优先使用租户指定模型
    - 意图路由: 不同意图使用不同模型
    - 负载均衡: 高峰期自动分流
    - 故障转移: 主模型不可用时自动切换
    """

    async def route(
        self,
        tenant_id: str,
        intent: str = None,
        priority: str = "normal"
    ) -> LLMAdapter:
        # 1. 检查租户配置
        tenant_config = await self.get_tenant_config(tenant_id)
        if tenant_config.preferred_model:
            adapter = self.get_adapter(tenant_config.preferred_model)
            if await self.check_availability(adapter):
                return adapter

        # 2. 意图路由
        if intent:
            model = self.intent_model_mapping.get(intent)
            if model:
                return self.get_adapter(model)

        # 3. 负载均衡选择
        return await self.load_balancer.select()
```

## 七、风险与应对

### 7.1 技术风险

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| LangGraph集成复杂度 | 对话流程延期 | 中 | 预留buffer时间，准备降级方案 |
| Milvus多租户性能 | RAG响应慢 | 中 | 提前进行压测，准备分库方案 |
| 微信支付审核 | 支付上线延期 | 高 | 提前提交申请，准备银行转账备选 |

| 风险 | 影响 | 概率 | 应对措施 |
